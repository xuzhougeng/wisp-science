import pandas as pd

RAW = "methods_raw.csv"
JMAP = "journal_map.json"

def build():
    df = pd.read_csv(RAW)
    jmap = pd.read_json(JMAP, typ="series")
    df["journal"] = df["journal_id"].map(jmap)
    cols = ["method", "year", "journal", "first_author"]
    out = df[cols].sort_values("year")
    out.to_csv("data-1.csv", index=False)
    return out
